860 requires std::derived_from<NodeType, Node>
861 gc<NodeType> Node::getParentNodeOfType(const std::string& sParentNodeName) {
862 std::scoped_lock guard(mtxParentNode.first);
863
864 // Check if have a parent.
865 if (mtxParentNode.second == nullptr) {
866 return nullptr;
867 }
868
869 // Check parent's type and optionally name.
870 if (dynamic_cast<NodeType*>(&*mtxParentNode.second) != nullptr &&
871 (sParentNodeName.empty() || mtxParentNode.second->getNodeName() == sParentNodeName)) {
872 // Found the node.
873 return gc_dynamic_pointer_cast<NodeType>(mtxParentNode.second);
874 }
875
876 // Ask parent nodes of that node.
877 return mtxParentNode.second->getParentNodeOfType<NodeType>(sParentNodeName);
878 }

880 template <typename NodeType>
881 requires std::derived_from<NodeType, Node>
882 gc<NodeType> Node::getChildNodeOfType(const std::string& sChildNodeName) {
883 std::scoped_lock guard(mtxChildNodes.first);
884
885 // Iterate over child nodes.
886 for (auto& pChildNode : *mtxChildNodes.second) {
887 // Check if this is the node we are looking for.
888 if (dynamic_cast<NodeType*>(&*pChildNode) != nullptr &&
889 (sChildNodeName.empty() || pChildNode->getNodeName() == sChildNodeName)) {
890 // Found the node.
891 return gc_dynamic_pointer_cast<NodeType>(pChildNode);
892 }
893
894 // Ask child nodes of that node.
895 const auto pNode = pChildNode->getChildNodeOfType<NodeType>(sChildNodeName);
896 if (pNode == nullptr) {
897 // Check the next child node.
898 continue;
899 }
900
901 // Found the node.
902 return pNode;
903 }
904
905 return nullptr;
906 }
